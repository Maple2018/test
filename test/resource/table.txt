CREATE TABLE `classroom` (
  `room_id` varchar(36) NOT NULL COMMENT '主键id',
  `campus_id` bigint(20) DEFAULT NULL COMMENT '校区id',
  `buliding_id` bigint(20) DEFAULT NULL COMMENT '楼栋id',
  `floor_id` bigint(20) DEFAULT NULL COMMENT '楼层id',
  `room_name` varchar(64) DEFAULT NULL COMMENT '教室名称',
  `model_id` bigint(20) DEFAULT NULL COMMENT '模板id',
  `open_days` char(1) DEFAULT NULL COMMENT '开放天数(0全周开放,1周末不开放,2仅周末开放)',
  `open_start_time` varchar(100) CHARACTER SET utf8 COLLATE utf8_bin DEFAULT NULL COMMENT '开放开始时段',
  `close_end_time` varchar(100) CHARACTER SET utf8 COLLATE utf8_bin DEFAULT NULL COMMENT '开放结束时段',
  `max_contain` int(11) DEFAULT NULL COMMENT '最大容纳',
  `hardware` varchar(420) DEFAULT NULL COMMENT '硬件设施',
  `description` varchar(420) DEFAULT NULL COMMENT '功能描述',
  `status` char(1) DEFAULT NULL COMMENT '状态(0有效,1无效)',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `create_user_id` bigint(20) DEFAULT NULL COMMENT '创建人',
  `update_time` datetime DEFAULT NULL COMMENT '修改时间',
  `update_user_id` bigint(20) DEFAULT NULL COMMENT '修改人',
  `come_from` char(1) DEFAULT NULL COMMENT '数据来源(1PC、2微信、3安卓、4iOS、5平板、6外部某系统)',
  `version` int(11) DEFAULT NULL COMMENT '版本',
  `org_id` varchar(36) DEFAULT NULL COMMENT '所属机构',
  `beizhu` varchar(420) DEFAULT NULL COMMENT '备注',
  PRIMARY KEY (`room_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='课室表';
